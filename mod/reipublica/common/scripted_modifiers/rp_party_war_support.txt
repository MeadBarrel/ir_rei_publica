rp_voting_supports = {
	modifier = {
		add = {
			value = 1
			divide = rp_$PARTY$_voting_size
		}
		var:$PARTY$_member_$INDEX$ = {
			custom_tooltip = {
				can_vote = yes
				rp_voting_$MATTER$_ruler > 0
				text = "rp_character_decision_support"
			}
		}
		
	}


	modifier = {
		add = {
			value = 1
			divide = rp_$PARTY$_voting_size
		}
		var:$PARTY$_member_$INDEX$ = {
			custom_tooltip = {
				can_vote = yes
				rp_voting_$MATTER$_ruler < 0
				text = "rp_character_decision_oppose"
			}
		}
	}


	modifier = {
		add = 0
		var:$PARTY$_member_$INDEX$ = {
			custom_tooltip = {
				can_vote = yes
				rp_voting_$MATTER$_ruler = 0
				text = "rp_character_decision_absolve"
			}
		}
		
	}

}
