rp_pi_pref_tribute = {
	value = 0
	add = {
		value = rp_get_political_ideology_elitarian
		subtract = rp_get_political_ideology_egalitarian
		divide = 2
	}
	value = rp_get_political_ideology_oppression
	subtract = rp_get_political_ideology_cooperation
}


rp_pref_pi_tribute_scaled = {
	if = {
		limit = { rp_pi_pref_tribute > 0 }
		value = rp_pi_pref_tribute
		divide = { value = rp_pi_pref_tribute multiply = rp_im add = rp_ia }	
	} 
	else = {
		value = rp_pi_pref_tribute
		divide = { value = rp_pi_pref_tribute multiply = rp_im_neg add = rp_ia }	
	}
}


rp_voting_tribute_personal = {
	value = rp_pref_pi_tribute_scaled
}



rp_voting_tribute_ruler = {
	value = rp_voting_tribute_personal
	add = rp_voting_support_ruler
}


rp_voting_tribute_military_party_ruler = {
	value = 0
	every_in_list = {
		variable = military_party_members
		limit = { can_vote = yes }
		if = { limit={ rp_voting_tribute_ruler>0 } add=rp_character_voting_power }
		else = { add={value=rp_character_voting_power multiply=-1} }
	}
}
