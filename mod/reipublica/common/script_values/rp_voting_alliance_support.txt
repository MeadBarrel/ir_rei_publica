rp_pref_pi_alliance = {
	value = 0
	add = rp_get_political_ideology_cooperation
	subtract = rp_get_political_ideology_oppression
	add = rp_get_political_ideology_openness 
	subtract = rp_get_political_ideology_isolation
}


rp_pref_pi_alliance_scaled = {
	if = {
		limit = { rp_pi_pref_alliance > 0 }
		value = rp_pi_pref_alliance
		divide = { value = rp_pi_pref_alliance multiply = rp_im add = rp_ia }	
	} 
	else = {
		value = rp_pi_pref_alliance
		divide = { value = rp_pi_pref_alliance multiply = rp_im_neg add = rp_ia }	
	}
}


rp_voting_alliance_personal = {
	value = rp_pref_pi_alliance_scaled

	add = 1  # Alliance is always good, no?

	if = {  # Zealous/pious and they have wrong religion
		limit = {
			or = {
				has_trait = zealous
				has_trait = pious
			}
			not = { has_same_religion_as = scope:target.current_ruler }
		}
		add = -5
	}

	if = {  # Has loyal friend in their country
		limit = {
			any_friend = {
				employer = scope:target
				loyalty >= 70
			}
		}
		add = 5
	}

	if = {  # Is friend of their ruler
		limit = { is_friend = scope:target.current_ruler }
		add = 10
	}


	if = {  # Has loyal rival in their country
		limit = {
			any_rival = {
				employer = scope:target
				loyalty > 70
			}
		}
		add = -5
	}

	if = {  # Their ruler is our rival
		limit = {
			is_rival = scope:target.current_ruler
		}
		add = -20
	}

	if = {  # We are not of the main culture and the target's ruler has the same culture as ours
		limit = {
			not = { has_same_culture_as = employer.current_ruler }
			has_same_culture_as = scope:target.current_ruler
		}
		add = 3
	}


	# From traits ======================================================================================================
	if = { limit = { has_trait = proud } add = -4 }  # Aren't we ourselves strong enough?
	if = { limit = { has_trait = suspicious } add = -4 }  # We shouldn't trust them!

}



rp_voting_alliance_ruler = {
	value = rp_voting_alliance_personal
	add = rp_voting_support_ruler
}


rp_voting_alliance_military_party_ruler = {
	value = 0
	every_in_list = {
		variable = military_party_members
		limit = { can_vote = yes }
		if = { limit={ rp_voting_alliance_ruler>0 } add=rp_character_voting_power }
		else = { add={value=rp_character_voting_power multiply=-1} }
	}
}

rp_voting_alliance_opinion_value = {
	if = {
		limit = { opinion = { target = scope:target value > 150 } }
		value = 0.09
	}
	else_if = {
		limit = { opinion = { target = scope:target value > 100 } }
		value = 0.06
	}
	else_if = {
		limit = { opinion = { target = scope:target value > 50 } }
		value = 0.03
	}	
	else_if = {
		limit = { opinion = { target = scope:target value < -150 } }
		value = -0.09
	}	
	else_if = {
		limit = { opinion = { target = scope:target value < -100 } }
		value = -0.06
	}	
	else_if = {
		limit = { opinion = { target = scope:target value < -50 } }
		value = -0.03
	}	
}
